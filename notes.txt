example of a double:
	//with a double its like a float except you can store 15-16 digits after the number
	double pi = 3.14159265358979;
	double e = 2.7182818284590;

	printf("The value of pi is %.14lf\n", pi);
	printf("The value of e is %.13lf\n", e);
	// '%lf' means long float, which is what a double is, you also have to specify how many numbers
	// after you want to print or it WILL round up, doubles are used for high precision numbers

example of a char:
	//Two things, 1. Char means character. 2. With characters use single quotes
	char grade = 'A';
	char symbol = '!';
	char currency = '$';
	//char's store single characters/symbols
	printf("Your grade is %c\n", grade);
	// %c is the one to call a character
	printf("Your favorite symbol is %c\n", symbol);
	printf("The currency is %c\n", currency);

example of char arrays (strings in C):
	//to store a string in C you need to use a char with an array
	//use double quotes for a array
	char name[] = "Lux/Roman";
	char food[] = "apples";
	char email[] = "fake123@gmail.com";
	
	// Use the format specifier %s for strings
	printf("Hello %s\n", name);
	printf("Your favorite food is %s\n", food);
	printf("Your email is %s\n", email);
